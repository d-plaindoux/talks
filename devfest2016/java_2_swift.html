Fonction :: Ordre Supèrieur

Fonction que prend une fonction comme entrée ou renvoie une fonction. 

let appliquer = { (f:(Int) -> Int, a:Int) -> Int in f(a) }
	    

func appliquer(_ f:(Int) -> Int, _ a:Int) -> Int {
  return f(a)
}
	    

appliquer(multiplier(2),3)

Fonction :: Généricité

Syntaxe "similaire" à celle adoptée par C++, Java et C# 

func f<A>(_ a:A) -> A { return a } // Identité
	    

func apply<A,B>(_ f : (A) -> B, a:A) -> B { return f(a) }

Fonction :: Exemple

Curryfication: Transforme une fonction à plusieurs arguments
en une fonction à un argument qui retourne un fonction 

func curry<A,B,C>(_ f: @escaping (A,B) -> C) -> (A) -> (B) -> C {
   return { a in { b in f(a,b) } } 
}

func uncurry(_ f: @escaping (A) -> (B) -> C) -> (A,B) -> C {
   return { a,b in f(a)(b) }
}	    

let multiplierCurry   : (Int) -> (Int) -> Int = curry(multiplier) 
let multiplierUncurry : (Int,Int) -> Int = uncurry(multiplierCurry)

Fonction :: η-Conversion 


  { a in f(a) } ≡ f 
	    

{ a in multiplierCurry(2)(a) } ≡ multiplierCurry(2)
Principe non préservé lors de l'utilisation de nom externe d'argument
