Transformation de fonction par curryfication et son inverse 

func curry<A,B,C>(_ f: @escaping (A,B) -> C) -> (A) -> (B) -> C {
   return { a in { b in f(a,b) } } 
}

func uncurry<A,B,C>(_ f: @escaping (A) -> (B) -> C) -> (A,B) -> C {
   return { a,b in f(a)(b) }
}	    

let multiplierCurry   : (Int) -> (Int) -> Int = curry(multiplier) 
let multiplierUncurry : (Int,Int) -> Int = uncurry(multiplierCurry)

Classe :: Initialisation optionnelle 


public class Personne {
  private let nom:String
  private var age:Int

  public init(nom:String, age:Int) {
    self.nom = nom
    self.age = age
  }

  convenience public init?(nom:String, anneeDeNaissance:Int) {
    if anneeCourante() < anneeDeNaissance { return nil }
    self.init(nom:nom, age:anneeCourante() - anneeDeNaissance)
  }
}
	    

let personne:Personne? = Personne(nom:"J. Doe", anneeDeNaissance:1974)
