Ma Permière Classe

class Personne {
 var nom : String = "Anonyme"
 var age : Int    = 0
}
var personne = Personne()
personne.nom = "John Doe"
personne.age = 42

Type synthétisé pour la variable personne

Attributs nom et age sont modifiables

Constructeur de classe

class Personne {
 let nom : String
 var age : Int

 init(nom:String, age:Int) {
  self.nom = nom
  self.age = age
 }
}
var personne = Personne(nom:"John Doe", age:42)
Attribut nom est immutable

Constructeurs de classe

class Personne {
 let nom : String
 var age : Int

 init(nom:String, age:Int) {
  self.nom = nom
  self.age = age
 }

 init(nom:String, année:Int) {
  self.nom = nom
  self.age = age:annéeCourante() - année
 }

Chaînage de constructeurs

class Personne {
 let nom : String
 var age : Int

 init(nom:String, age:Int) {
  self.nom = nom
  self.age = age
 } 

 convenience init(nom:String, année:Int) {
  self.init(nom:nom, age:annéeCourante() - année)
 }
}

Constructeur et cas limite

class Personne {
 let nom : String
 var age : Int

 init(nom:String, age:Int) {
  self.nom = nom
  self.age = age
 } 

 convenience init?(nom:String, année:Int) {
  if annéeCourante() - année < 0 { return nil }
  self.init(nom:nom, age:annéeCourante() - année)
 }
}
